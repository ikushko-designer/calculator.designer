<script>
function calculate() {
  const placement = parseFloat(document.getElementById("placement").value);
  const selectedCity = document.getElementById("city").value;
  const budgetMin = parseFloat(document.getElementById("budgetMin").value) || 0;
  const budgetMax = parseFloat(document.getElementById("budgetMax").value) || Infinity;
  const sortOrder = document.getElementById("sort").value;

  function generateDesignerCard(name, id, customRates=false) {
    let total = 0;
    let html = `
    <div class="designer">
      <div class="header">
        <div class="designer-photo">
          <img src="https://images.unsplash.com/photo-1544005313-94ddf0286df2?w=400">
        </div>
        <div class="header-info">
          <h3>${name} ${id} <span class="city">УФА</span></h3>
          <div class="tags">
            <span>Может выехать на объект</span>
            <span>Составляет ТЗ онлайн</span>
            <span>Дом</span>
            <span>Квартира</span>
          </div>
        </div>
      </div>
      <h4>Расчет стоимости:</h4>
    `;

    function row(name, val, rate, amount) {
      html += `
      <div class="calc-row">
        <div class="service-line">
          <span>${name}</span>
          <span class="rate">${rate}</span>
        </div>
        <div class="amount-box">
          <span>${amount}</span>
          <span>${Math.round(val)} ₽</span>
        </div>
      </div>`;
    }

    const rateMeasure = customRates ? 90 : 110;
    const rateDocs = customRates ? 90 : 490;
    const rateVisual = customRates ? 90 : 700;
    const rateTable = customRates ? 90 : 160;
    const rateCollage = customRates ? 90 : 1500;

    if (measure.checked) { let val = area1.value * rateMeasure * placement; total += val; row("Обмер", val, rateMeasure + " ₽/м²", area1.value + " кв.м"); }
    if (docs.checked) { let val = area2.value * rateDocs * placement; total += val; row("Документация", val, rateDocs + " ₽/м²", area2.value + " кв.м"); }
    if (visual.checked) { let val = area3.value * rateVisual * placement; total += val; row("3D-визуализация", val, rateVisual + " ₽/м²", area3.value + " кв.м"); }
    if (table.checked) { let val = areaTable.value * rateTable; total += val; row("Таблица комплектации", val, rateTable + " ₽/м²", areaTable.value + " кв.м"); }
    if (collage.checked) { let val = collageCount.value * rateCollage; total += val; row("2D-коллажи", val, rateCollage + " ₽/ракурс", collageCount.value + " ракурс(ов)"); }

    html += `<div class="total">Итого: ${Math.round(total)} ₽</div>`;
    html += `<button class="contact-btn">Связаться</button>`;
    html += `</div>`;
    return { html, total };
  }

  // создаем массив дизайнеров
  const designers = [
    generateDesignerCard("Полина", "0001758"),
    generateDesignerCard("Екатерина", "0001759", true)
  ];

  // фильтруем по бюджету и городу
  let filteredDesigners = designers.filter(d => d.total >= budgetMin && d.total <= budgetMax && selectedCity === "Уфа");

  // сортировка
  if (sortOrder === "asc") filteredDesigners.sort((a,b) => a.total - b.total);
  else if (sortOrder === "desc") filteredDesigners.sort((a,b) => b.total - a.total);

  // вывод
  document.getElementById("results").innerHTML =
    filteredDesigners.length > 0 ? filteredDesigners.map(d => d.html).join("") : "<p>Нет дизайнеров по выбранным параметрам</p>";
}
</script>
